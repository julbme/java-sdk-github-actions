package me.julb.sdk.github.actions.spi;

import java.util.Optional;
import java.util.ServiceLoader;

import lombok.AccessLevel;
import lombok.RequiredArgsConstructor;

/**
 * The Service Loader helper used to load a GitHub action. <br>
 * @author Julb.
 */
@RequiredArgsConstructor(access = AccessLevel.PRIVATE)
public class GitHubActionServiceLoader {
    /**
     * Gets the GitHub action provider implementation using Java SPI.
     * @return the first {@link GitHubActionProvider} found or {@link Optional#empty()} if not defined.
     */
    public static Optional<GitHubActionProvider> getImplementation() {
        return ServiceLoader.load(GitHubActionProvider.class).findFirst();
    }
}
